Count Excel Row
    [Arguments]    ${Number}
    ${Value}    Read Excel Row    row_num=${Number}
    Remove Values From List    ${Value}    ${None}
    log list    ${Value}
    ${Legth_Row}    Get Length    ${Value}
    log    ${Legth_Row}
    [Return]    ${Legth_Row}

Count Excel Column
    [Arguments]    ${Number}
    ${Value}    Read Excel Column    col_num=${Number}
    Remove From List    ${Value}    0
    log list    ${Value}
    ${Legth_Column}    Get Length    ${Value}
    log    ${Legth_Column}
    [Return]    ${Legth_Column}
